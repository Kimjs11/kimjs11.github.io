import os

    os.makedirs(new_txt_path)

# if not os.path.exists(image_copy_path):

    for jpg_file in files:
        if not jpg_file.endswith('.jpg'):
            continue
        file_name = jpg_file.split('.jpg')[0]+'.txt'
#         print(file_name)
        
#         print(new_path)
        file_images = txt_path+'/'+file_name
